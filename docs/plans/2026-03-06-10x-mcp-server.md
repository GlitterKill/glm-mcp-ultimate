> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Transform the GLM MCP server into a production-grade, plan-centric orchestration platform that enables LLMs to leverage GLM models as intelligent agents with streaming, parallel execution, and structured feedback.

**Architecture:** Layered architecture with: (1) Auth abstraction layer (API key + OAuth), (2) Plan parser and executor, (3) Agent pool with sandboxed execution, (4) Streaming event bus, (5) Session persistence with SQLite, (6) Structured feedback system.

```typescript
// src/auth/oauth-provider.ts
import type { AuthProvider, AuthConfig } from "./types.js";

export class OAuthProvider implements AuthProvider {
  private accessToken?: string;
  private expiresAt?: number;
  private refreshToken?: string;

  constructor(config: NonNullable<AuthConfig["oauth"]>) {
    this.accessToken = config.accessToken;
    this.expiresAt = config.expiresAt;
    this.refreshToken = config.refreshToken;
  }

  async getAccessToken(): Promise<string> {
    if (this.isTokenExpired()) {
      await this.refresh();
    }
    if (!this.accessToken) {
      throw new Error("No access token available");
    }
    return this.accessToken;
  }

  getType(): "oauth" {
    return "oauth";
  }

  private isTokenExpired(): boolean {
    if (!this.expiresAt) return !this.accessToken;
    return Date.now() >= this.expiresAt - 60000;
  }

  async refresh(): Promise<void> {
    // OAuth refresh implementation
  }
}
```

```typescript
// src/auth/factory.ts
import type { AuthConfig, AuthProvider } from "./types.js";
import { ApiKeyProvider } from "./api-key-provider.js";
import { OAuthProvider } from "./oauth-provider.js";

export function createAuthProvider(config: AuthConfig): AuthProvider {
  switch (config.type) {
    case "api_key":
      if (!config.apiKey) throw new Error("API key required");
      return new ApiKeyProvider(config.apiKey);
    case "oauth":
      if (!config.oauth) throw new Error("OAuth config required");
      return new OAuthProvider(config.oauth);
    default:
      throw new Error(`Unknown auth type`);
  }
}

export function authConfigFromEnv(): AuthConfig {
  const apiKey = process.env.GLM_API_KEY;
  if (apiKey) return { type: "api_key", apiKey };

  const clientId = process.env.GLM_OAUTH_CLIENT_ID;
  const clientSecret = process.env.GLM_OAUTH_CLIENT_SECRET;
  if (clientId && clientSecret) {
    return {
      type: "oauth",
      oauth: {
        clientId,
        clientSecret,
        tokenUrl: process.env.GLM_OAUTH_TOKEN_URL || "https://api.z.ai/oauth/token",
        refreshToken: process.env.GLM_OAUTH_REFRESH_TOKEN,
        accessToken: process.env.GLM_OAUTH_ACCESS_TOKEN,
        expiresAt: process.env.GLM_OAUTH_EXPIRES_AT 
          ? parseInt(process.env.GLM_OAUTH_EXPIRES_AT, 10) 
          : undefined,
      },
    };
  }

  throw new Error("No valid authentication configuration found");
}
```
